



CREATE TABLE Companies (
	CompanyId int PRIMARY KEY NOT NULL ,
	CompanyName varchar (255) NOT NULL ,
	City varchar(30) NOT NULL ,
	Streetname varchar(30) NOT NULL ,
	Postal_code int NOT NULL ,
	CompanyTelephone varchar (20) NOT NULL 
) 

;

CREATE TABLE Contacts (
	ContactId int PRIMARY KEY NOT NULL ,
	ContactName varchar (40) NOT NULL ,
	ContactFirstname varchar (40) NOT NULL ,
	ContactTelephone varchar (20) NOT NULL ,
	ContactEmail varchar (30) NOT NULL ,
	CompanyId int NOT NULL,
	CONSTRAINT FK_CompanyId FOREIGN KEY (CompanyId) REFERENCES Companies(CompanyId)

) ;

CREATE TABLE Students (
	StudentId int PRIMARY KEY NOT NULL ,
	StudentName varchar (40) NOT NULL ,
	StudentFirstName varchar (40) NOT NULL ,
	Departement varchar(5) NOT NULL ,
	Studenttelephone varchar (20) NOT NULL ,
	Studentemail varchar (30) NOT NULL ,
	Documents BYTEA
) 
;
CREATE TABLE Internships
(
	CompanyId int  NOT NULL,
	StudentId int NOT NULL,
	year date,

	CONSTRAINT FK_CompanyId FOREIGN KEY (CompanyId) REFERENCES Companies(CompanyId),
	CONSTRAINT FK_StudentId FOREIGN KEY (StudentId) REFERENCES Students(StudentId),
	CONSTRAINT InternshipId PRIMARY KEY (CompanyId, StudentId)


) ;
